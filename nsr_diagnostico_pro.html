// ────────────────────────────────────────────── // NSR Diagnóstico Pro · Apps Script Receiver // Pega esto en: Extensiones → Apps Script // ────────────────────────────────────────────── function doPost(e) { try { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); if (data.tipo === 'prospecto') { var sh = ss.getSheetByName('Prospectos') || ss.insertSheet('Prospectos'); if (sh.getLastRow() === 0) { sh.appendRow(['ID','Negocio','Giro','Dueño','Teléfono','Dirección','Zona','Estatus','Fecha Alta','Notas']); sh.getRange(1,1,1,10).setBackground('#0D1B3E').setFontColor('#C9A84C').setFontWeight('bold'); } sh.appendRow([data.id, data.negocio, data.giro, data.dueno, data.telefono, data.direccion, data.zona, data.estatus, data.fechaAlta, data.notas]); } if (data.tipo === 'diagnostico') { var sh2 = ss.getSheetByName('Diagnosticos') || ss.insertSheet('Diagnosticos'); if (sh2.getLastRow() === 0) { sh2.appendRow(['ID','Negocio','Giro','Fecha','Consultor','Score General','Ventas','Inventario','Finanzas','Procesos','Personas','Tecnología','Oportunidades','Notas','Hallazgos']); sh2.getRange(1,1,1,15).setBackground('#0D1B3E').setFontColor('#C9A84C').setFontWeight('bold'); } sh2.appendRow([ data.id, data.negocio, data.giro, data.fecha, data.consultor, data.scoreGeneral, data.areas.ventas, data.areas.inventario, data.areas.finanzas, data.areas.procesos, data.areas.personas, data.areas.tecnologia, data.oportunidades, data.notas, data.hallazgos ]); } return ContentService.createTextOutput(JSON.stringify({ok: true})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({ok: false, error: err.toString()})) .setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status: 'NSR Diagnóstico Pro · Online ✅'})) .setMimeType(ContentService.MimeType.JSON); }
